    public function __replace($content, $keyword)
    {
        $opts = array("before_match"=>"<font style='font-weight:bold;color:#f00'>","after_match"=>"</font>");

        $keyword = preg_split('/(?<!^)(?!$)/u', $keyword );

        foreach ($keyword as $word) {
            $content = str_replace($word,$opts['before_match'] . $word . $opts['after_match'],$content);
        }

        return $content;
    }
